Botón que cambia de estilo

    
    <body>

      <h2>Botón que cambia de estilo</h2>
      <button id="btn-estilo">Haz clic para cambiar estilo</button>

      <script>

        const boton = document.getElementById('btn-estilo');



        boton.addEventListener('click', function () {

          // Si el botón aún no tiene el estilo cambiado

          if (this.style.backgroundColor !== 'green') {

              this.style.backgroundColor = 'green';

              this.style.color = 'white';

              this.style.fontSize = '20px';

              this.textContent = 'Estilo aplicado';

          } else {

              // Volver al estilo original

              this.style.backgroundColor = 'lightgray';

              this.style.color = 'black';

              this.style.fontSize = '16px';

              this.textContent = 'Haz clic para cambiar estilo';

          }

        });

      </script>

    </body>
